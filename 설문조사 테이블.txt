create table survey ( 
	num number constraint survey_num_pk primary key, --설문조사의 고유 번호
	id varchar2(50), --설문조사를 한 사람의 id
	code number, --설문 구분 코드
	sub varchar2(255), --설문 제목
	sdate date, --설문조사 실시 날짜
	scheck boolean NOT NULL, --설문조사 실시 여부
	constraint survey_id_FK FOREIGN KEY (id) references member(id)
);

create sequence survey_seq increment by 1 start with 1;

create table surveycontent (
	subcode number(19) NOT NULL, --설문조사 컨텐츠가 들어갈 설문조사의 번호
	surveycnt number(10), -- 설문조사를 실시한 횟수
	surveytitle varchar2(255), --설문조사 제목
	surveytype varchar2(255), --
	contraint surveycontent_subcode_FK FOREIGN KEY (subcode) references survey(num)
);
